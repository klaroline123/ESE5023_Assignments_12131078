import numpy as np
import itertools as itls
import matplotlib.pyplot as plt

def Find_expression(value, i=0): # 用于找到表达式和表达式数量，并打印出来。
    array = itls.product(['+','-',''], repeat=8) # 枚举法
    for ops in array:
        expr = f"1{ops[0]}2{ops[1]}3{ops[2]}4{ops[3]}5{ops[4]}6{ops[5]}7{ops[6]}8{ops[7]}9"
        if eval(expr) == value: # eval() 函数用来执行一个字符串表达式，并返回表达式的值。
            i+=1
            print(expr, f"={value}")
    print("total num of expr is: " + str(i))
    return i

def Find_expression_num(value, i=0): # 用于找到表达式的数量
    array = itls.product(['+','-',''], repeat=8)
    for ops in array:
        expr = f"1{ops[0]}2{ops[1]}3{ops[2]}4{ops[3]}5{ops[4]}6{ops[5]}7{ops[6]}8{ops[7]}9"
        if eval(expr) == value: # eval() 函数用来执行一个字符串表达式，并返回表达式的值。
            i+=1
    return i

def find_num(item, count_list): # 用于找出目标数字的在list中的位置
    idx_list = []
    for index, element in enumerate(count_list):
        if element == item:
            idx_list.append(index+1)
#             print(index+1, end=" ")
    return idx_list
